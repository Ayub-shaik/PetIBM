82 {

84

88  Vec RDiag, MHatDiag; // a temporary vectors

91  // create basic operators
92  ierr = operators::createDivergence(
93  mesh, bc, DE[0], DCorrection, PETSC_FALSE); CHKERRQ(ierr);
94  ierr = operators::createGradient(mesh, GH[0], PETSC_FALSE); CHKERRQ(ierr);
95  ierr = operators::createLaplacian(mesh, bc, L, LCorrection); CHKERRQ(ierr);
96  ierr = operators::createConvection(mesh, bc, N); CHKERRQ(ierr);
97 
98  // create combined operator: A
99  ierr = MatDuplicate(L, MAT_COPY_VALUES, &A); CHKERRQ(ierr);
100  ierr = MatScale(A, -diffCoeffs->implicitCoeff * nu); CHKERRQ(ierr);
101  ierr = MatShift(A, 1.0/dt); CHKERRQ(ierr);
102 
103  // create diagonal matrix R and get a Vec holding the diagonal
104  ierr = petibm::operators::createR(mesh, R); CHKERRQ(ierr);
105  ierr = MatCreateVecs(R, nullptr, &RDiag); CHKERRQ(ierr);
106  ierr = MatGetDiagonal(R, RDiag); CHKERRQ(ierr);
107 
108  // create diagonal matrix MHat and get a Vec holding the diagonal
109  ierr = petibm::operators::createMHead(mesh, MHat); CHKERRQ(ierr);
110  ierr = MatCreateVecs(MHat, nullptr, &MHatDiag); CHKERRQ(ierr);
111  ierr = MatGetDiagonal(MHat, MHatDiag); CHKERRQ(ierr);
112 
113  // create a Delta operator and its transpose (equal to H)
114  ierr = petibm::operators::createDelta(mesh, bc, bodies, DE[1]); CHKERRQ(ierr);
115  ierr = MatTranspose(DE[1], MAT_INITIAL_MATRIX, &GH[1]); CHKERRQ(ierr);
116 
117  // create operator E
118  ierr = MatDiagonalScale(DE[1], nullptr, RDiag); CHKERRQ(ierr);
119  ierr = MatDiagonalScale(DE[1], nullptr, MHatDiag); CHKERRQ(ierr);
120  ierr = VecDestroy(&RDiag); CHKERRQ(ierr);
121  ierr = VecDestroy(&MHatDiag); CHKERRQ(ierr);
122  ierr = MatDestroy(&R); CHKERRQ(ierr);
123  ierr = MatDestroy(&MHat); CHKERRQ(ierr);
124 
125  // create -H (H operator in GH combination has a minus sign)
126  ierr = MatScale(GH[1], -1.0); CHKERRQ(ierr);
127 
128  // get combined operator G; R is used temporarily
129  ierr = MatCreateNest(mesh->comm, 1, nullptr, 2, nullptr, GH, &R); CHKERRQ(ierr);
130  ierr = MatConvert(R, MATAIJ, MAT_INITIAL_MATRIX, &G); CHKERRQ(ierr);
131  ierr = MatDestroy(&R); CHKERRQ(ierr);
132  ierr = MatDestroy(&GH[0]); CHKERRQ(ierr);
133  ierr = MatDestroy(&GH[1]); CHKERRQ(ierr);
134 
135  // get combined operator D; R is used temporarily; also get ISs
136  ierr = MatCreateNest(mesh->comm, 2, nullptr, 1, nullptr, DE, &R); CHKERRQ(ierr);
137  ierr = MatConvert(R, MATAIJ, MAT_INITIAL_MATRIX, &D); CHKERRQ(ierr);
138  ierr = MatNestGetISs(R, is, nullptr); CHKERRQ(ierr);
139  ierr = ISDuplicate(is[0], &isDE[0]); CHKERRQ(ierr);
140  ierr = ISDuplicate(is[1], &isDE[1]); CHKERRQ(ierr);
141  ierr = ISCopy(is[0], isDE[0]); CHKERRQ(ierr);
142  ierr = ISCopy(is[1], isDE[1]); CHKERRQ(ierr);
143  ierr = MatDestroy(&R); CHKERRQ(ierr);
144  ierr = MatDestroy(&DE[0]); CHKERRQ(ierr);
145  ierr = MatDestroy(&DE[1]); CHKERRQ(ierr);
146 
147  // create combined operator: BNG
148  ierr = operators::createBnHead(
149  L, dt, diffCoeffs->implicitCoeff * nu, 1, BN); CHKERRQ(ierr);
150  ierr = MatMatMult(
151  BN, G, MAT_INITIAL_MATRIX, PETSC_DEFAULT, &BNG); CHKERRQ(ierr);
152 
153  // create combined operator: DBNG
154  ierr = MatMatMult(
155  D, BNG, MAT_INITIAL_MATRIX, PETSC_DEFAULT, &DBNG); CHKERRQ(ierr);
156 
157  // set null space to DBNG
158  ierr = setNullSpace(); CHKERRQ(ierr);
159 
160  // destroy temporary operator
161  ierr = MatDestroy(&BN); CHKERRQ(ierr);
162 
163  PetscFunctionReturn(0);
164 } // createOperators

169 {

178
