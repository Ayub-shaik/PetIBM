294  PetscInt &nTotal, PetscReal &ed,
295  type::RealVec1D &dL)
296 {

329 PetscErrorCode parseOneSubDomain(const YAML::Node &sub, const PetscReal bg,
330  PetscInt &n, PetscReal &ed,
331  type::RealVec1D &dL)
332 {
333  PetscFunctionBeginUser;
334 
335  // get the number of the cells in this sub-domain
336  n = sub["cells"].as<PetscInt>();
337 
338  // get the end of the sub-domain
339  ed = sub["end"].as<PetscReal>();
340 
341  // get the stretching ratio
342  PetscReal r = sub["stretchRatio"].as<PetscReal>();
343 
344  // obtain the cell sizes
345  if (std::abs(r - 1.0) <= 1e-12)
346  dL = type::RealVec1D(n, (ed - bg) / n); // uniform grid
347  else
348  misc::stretchGrid(bg, ed, n, r, dL); // stretch grid
349 
350  PetscFunctionReturn(0);
