70  PetscErrorCode ierr;

72  DMDALocalInfo lclInfo;
73

75  nPts, dim, 0, nullptr, &da); CHKERRQ(ierr);
76 
77  ierr = DMSetUp(da); CHKERRQ(ierr);

90  // each point has "dim" degree of freedom, so we have to multiply that
91  for(auto &it: nLclAllProcs) it *= dim;
92 
93  // calculate the offset of the unpacked DM

95  for(PetscMPIInt r=1; r<mpiSize; r++) offsetsAllProcs[r] += offsetsAllProcs[r-1];

101 PetscErrorCode SingleBodyPoints::updateMeshIdx()
102 {
103  PetscFunctionBeginUser;
104 
105  // loop through points owned locally and find indices
106  for(PetscInt i=bgPt, c=0; i<edPt; ++i, ++c)
107  {
108  for(PetscInt d=0; d<dim; ++d)
109  {
110  if (mesh->min[d] >= coords[i][d] || mesh->max[d] <= coords[i][d])
111  {
112  SETERRQ3(PETSC_COMM_WORLD, PETSC_ERR_MAX_VALUE,
113  "body coordinate %g is outside domain [%g, %g] !",
114  coords[i][d], mesh->min[d], mesh->max[d]);
115  }
116 
117  meshIdx[c][d] = std::upper_bound(
118  mesh->coord[4][d], mesh->coord[4][d]+mesh->n[4][d],
119  coords[i][d]) - mesh->coord[4][d] - 1;
120  }
121  }

123  PetscFunctionReturn(0);
