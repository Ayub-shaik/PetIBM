174 PetscErrorCode NavierStokesSolver::createOperators()
175 {
176  PetscErrorCode ierr;
177 
178  PetscFunctionBeginUser;
179 
180  Mat BN; // a temporary operator
181 
182  // create basic operators
183  ierr = operators::createDivergence(
184  mesh, bc, D, DCorrection, PETSC_FALSE); CHKERRQ(ierr);
185  ierr = operators::createGradient(mesh, G, PETSC_FALSE); CHKERRQ(ierr);
186  ierr = operators::createLaplacian(mesh, bc, L, LCorrection); CHKERRQ(ierr);
187  ierr = operators::createConvection(mesh, bc, N); CHKERRQ(ierr);
188 
189  // create combined operator: A
190  ierr = MatDuplicate(L, MAT_COPY_VALUES, &A); CHKERRQ(ierr);
191  ierr = MatScale(A, -diffCoeffs->implicitCoeff * nu); CHKERRQ(ierr);
192  ierr = MatShift(A, 1.0/dt); CHKERRQ(ierr);
193 
194  // create combined operator: BNG
195  ierr = operators::createBnHead(
196  L, dt, diffCoeffs->implicitCoeff * nu, 1, BN); CHKERRQ(ierr);
197  ierr = MatMatMult(
198  BN, G, MAT_INITIAL_MATRIX, PETSC_DEFAULT, &BNG); CHKERRQ(ierr);
199 
200  // create combined operator: DBNG
201  ierr = MatMatMult(
202  D, BNG, MAT_INITIAL_MATRIX, PETSC_DEFAULT, &DBNG); CHKERRQ(ierr);
203 
204  // set null space to DBNG
205  ierr = setNullSpace(); CHKERRQ(ierr);
206 
207  // destroy temporary operator
208  ierr = MatDestroy(&BN); CHKERRQ(ierr);
209 
210  PetscFunctionReturn(0);

302 // prepare right-hand-side vector for velocity solver
303 PetscErrorCode NavierStokesSolver::assembleRHSVelocity()
304 {
305  PetscErrorCode ierr;
306 
307  PetscFunctionBeginUser;
308 
309  ierr = PetscLogStagePush(stageRHSVelocity); CHKERRQ(ierr);
310 
311  // initialize RHS with pressure gradient at time-step n
312  ierr = MatMult(G, solution->pGlobal, rhs1); CHKERRQ(ierr);
313  ierr = VecScale(rhs1, -1.0); CHKERRQ(ierr);
314 
315  // add explicit part of time derivative ($\frac{u^n}{\Delta t}$)
316  ierr = VecAXPY(rhs1, 1.0/dt, solution->UGlobal); CHKERRQ(ierr);
317 
318  // add convection terms from time index n, n-1, n-2, ... to rhs1:
319  {
320  // 1. discard the oldest-time-step term, and decrease the time-step by 1
321  for(int i=conv.size()-1; i>0; i--) {
322  ierr = VecSwap(conv[i], conv[i-1]); CHKERRQ(ierr);
323  }
324 
325  if (conv.size() > 0)
326  {
327  // 2. get convection term at time-step n
328  ierr = MatMult(N, solution->UGlobal, conv[0]); CHKERRQ(ierr);
329 
330  // 3. move n-th time-step convection term to right-hand-side
331  ierr = VecScale(conv[0], -1.0); CHKERRQ(ierr);
332  }
333 
334  // 4. add all explicit convective terms to rhs1
335  for(unsigned int i=0; i<conv.size(); ++i) {
336  ierr = VecAXPY(rhs1, convCoeffs->explicitCoeffs[i], conv[i]);
337  CHKERRQ(ierr);
338  }
339  }
340 
341  // add diffusion terms from time index n, n-1, n-2, ... to rhs1:
342  {
343  // 1. discard the oldest-time-step term, and decrease the time-step by 1
344  for(int i=diff.size()-1; i>0; i--) {
345  ierr = VecSwap(diff[i], diff[i-1]); CHKERRQ(ierr);
346  }
347 
348  // 2. get diffusion term at time-step n
349  if (diff.size() > 0)
350  {
351  ierr = MatMult(L, solution->UGlobal, diff[0]); CHKERRQ(ierr);
352  ierr = MatMultAdd(LCorrection,
353  solution->UGlobal, diff[0], diff[0]); CHKERRQ(ierr);
354  ierr = VecScale(diff[0], nu); CHKERRQ(ierr);
355  }
356 
357  // 3. add all explicit diffusive terms to rhs1
358  for(unsigned int i=0; i<diff.size(); ++i) {
359  ierr = VecAXPY(rhs1, diffCoeffs->explicitCoeffs[i], diff[i]);
360  CHKERRQ(ierr);
361  }
362  }
363 
364  // add the diffusion BC correction from time index n+1 to rhs1
365  {
366  // 1. update the Eq.s of ghost points to time index n+1
367  ierr = bc->updateEqs(solution, dt); CHKERRQ(ierr);
368 
369  // 2. get BC correction of Laplacian of time n+1
370  ierr = MatMult(LCorrection, solution->UGlobal, bc1); CHKERRQ(ierr);
371  ierr = VecScale(bc1, nu); CHKERRQ(ierr);
372 
373  // 3. add bc1 to rhs
374  ierr = VecAXPY(rhs1, diffCoeffs->implicitCoeff, bc1); CHKERRQ(ierr);
375  }
376 
377  ierr = PetscLogStagePop(); CHKERRQ(ierr);
378 
379  PetscFunctionReturn(0);
380 } // assembleRHSVelocity
