312 PetscErrorCode NavierStokesSolver::createOperators()
313 {
314  PetscErrorCode ierr;
315 
316  PetscFunctionBeginUser;
317 
318  Mat BN; // a temporary operator
319 
320  // create the divergence operator: D
321  ierr = petibm::operators::createDivergence(
322  mesh, bc, D, DCorrection, PETSC_FALSE); CHKERRQ(ierr);
323 
324  // create the gradient operator: G
325  ierr = petibm::operators::createGradient(
326  mesh, G, PETSC_FALSE); CHKERRQ(ierr);
327 
328  // create the Laplacian operator: L
329  ierr = petibm::operators::createLaplacian(
330  mesh, bc, L, LCorrection); CHKERRQ(ierr);
331 
332  // create the operator for the convective terms: N
333  ierr = petibm::operators::createConvection(
334  mesh, bc, N); CHKERRQ(ierr);
335 
336  // create the implicit operator of the velocity system: A
337  ierr = MatDuplicate(L, MAT_COPY_VALUES, &A); CHKERRQ(ierr);
338  ierr = MatScale(A, -diffCoeffs->implicitCoeff * nu); CHKERRQ(ierr);
339  ierr = MatShift(A, 1.0 / dt); CHKERRQ(ierr);
340 
341  // create the projection operator: BNG
342  ierr = petibm::operators::createBnHead(
343  L, dt, diffCoeffs->implicitCoeff * nu, 1, BN); CHKERRQ(ierr);
344  ierr = MatMatMult(
345  BN, G, MAT_INITIAL_MATRIX, PETSC_DEFAULT, &BNG); CHKERRQ(ierr);
346 
347  // create the Poisson operator: DBNG
348  ierr = MatMatMult(
349  D, BNG, MAT_INITIAL_MATRIX, PETSC_DEFAULT, &DBNG); CHKERRQ(ierr);
350 
351  // set the nullspace of the Poisson system
352  ierr = setNullSpace(); CHKERRQ(ierr);
353 
354  // destroy the temporary operator
355  ierr = MatDestroy(&BN); CHKERRQ(ierr);
356 
357  PetscFunctionReturn(0);
358 } // createOperators

424 // assemble the right-hand side vector of the velocity system
425 PetscErrorCode NavierStokesSolver::assembleRHSVelocity()
426 {
427  PetscErrorCode ierr;
428 
429  PetscFunctionBeginUser;
430 
431  ierr = PetscLogStagePush(stageRHSVelocity); CHKERRQ(ierr);
432 
433  // initialize RHS vector with pressure gradient at time-step n
434  // $rhs_1 = - \frac{\partial p^n}{\partial x}$
435  ierr = MatMult(G, solution->pGlobal, rhs1); CHKERRQ(ierr);
436  ierr = VecScale(rhs1, -1.0); CHKERRQ(ierr);
437 
438  // add explicit part of time derivative to the RHS vector
439  // $rhs_1 += \frac{u^n}{\Delta t}$
440  ierr = VecAXPY(rhs1, 1.0 / dt, solution->UGlobal); CHKERRQ(ierr);
441 
442  // add all explicit convective terms to the RHS vector
443  // $rhs_1 += \sum_{k=0}^s conv_{n - k}$
444  {
445  // 1. discard the term at the oldest time-step
446  // and decrease the time-step by 1
447  for (int i = conv.size() - 1; i > 0; i--)
448  {
449  ierr = VecSwap(conv[i], conv[i - 1]); CHKERRQ(ierr);
450  }
451 
452  if (conv.size() > 0)
453  {
454  // 2. compute and store the convective term from time-step n
455  ierr = MatMult(N, solution->UGlobal, conv[0]); CHKERRQ(ierr);
456 
457  // 3. scale the newest convective term by -1
458  // (to be added to the RHS vector)
459  ierr = VecScale(conv[0], -1.0); CHKERRQ(ierr);
460  }
461 
462  // 4. add all explicit convective terms to the RHS vector
463  for (unsigned int i = 0; i < conv.size(); ++i)
464  {
465  ierr = VecAXPY(rhs1, convCoeffs->explicitCoeffs[i], conv[i]);
466  CHKERRQ(ierr);
467  }
468  }
469 
470  // add all explicit diffusion terms to the RHS vector
471  // $rhs_1 += \sum_{k=0}^s diff_{n - k}$
472  {
473  // 1. discard the term at the oldest time-step
474  // and decrease the time-step by 1
475  for (int i = diff.size() - 1; i > 0; i--)
476  {
477  ierr = VecSwap(diff[i], diff[i - 1]); CHKERRQ(ierr);
478  }
479 
480  // 2. compute and store the diffusion term from time-step n
481  if (diff.size() > 0)
482  {
483  ierr = MatMult(L, solution->UGlobal, diff[0]); CHKERRQ(ierr);
484  ierr = MatMultAdd(
485  LCorrection, solution->UGlobal, diff[0], diff[0]); CHKERRQ(ierr);
486  ierr = VecScale(diff[0], nu); CHKERRQ(ierr);
487  }
488 
489  // 3. add all explicit diffusion terms to the RHS vector
490  for (unsigned int i = 0; i < diff.size(); ++i)
491  {
492  ierr = VecAXPY(
493  rhs1, diffCoeffs->explicitCoeffs[i], diff[i]); CHKERRQ(ierr);
494  }
495  }
496 
497  // add implicit BC correction terms arising from the diffusion term
498  // to the RHS vector
499  {
500  // 1. update the ghost-point equations
501  ierr = bc->updateEqs(solution, dt); CHKERRQ(ierr);
502 
503  // 2. compute the implicit BC correction terms
504  ierr = MatMult(LCorrection, solution->UGlobal, bc1); CHKERRQ(ierr);
505  ierr = VecScale(bc1, nu); CHKERRQ(ierr);
506 
507  // 3. add the correction terms to the RHS vector
508  ierr = VecAXPY(rhs1, diffCoeffs->implicitCoeff, bc1); CHKERRQ(ierr);
509  }
510 
511  ierr = PetscLogStagePop(); CHKERRQ(ierr); // end of stageRHSVelocity
512 
513  PetscFunctionReturn(0);
514 } // assembleRHSVelocity
