172  const type::Mesh &mesh, const PetscInt &dof, const type::IntVec1D &IJK,
173  const std::vector<bool> &periodic, const PetscInt &window,

187  ijk[d].push_back(s);
188  xyz[d].push_back(mesh->coord[dof][d][s]);

197  }
