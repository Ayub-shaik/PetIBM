237 {

240  PetscFunctionBeginUser;
241

418 PetscErrorCode ProbeVolume::monitorVec(const DM &da,
419  const Vec &fvec,
420  const PetscInt &n,
421  const PetscReal &t)
422 {
423  PetscErrorCode ierr;
424 
425  PetscFunctionBeginUser;
426 
427  // grab the part of the vector that corresponds to the sub-volume
428  Vec svec;
429  ierr = VecGetSubVector(fvec, is, &svec); CHKERRQ(ierr);
430  if (n_sum != 0) // we accumulate the data over the time-steps
431  {
432  if (dvec == PETSC_NULL)
433  {
434  ierr = VecDuplicate(svec, &dvec); CHKERRQ(ierr);
435  ierr = VecSet(dvec, 0.0); CHKERRQ(ierr);
436  }
437  ierr = VecAXPY(dvec, 1.0, svec); CHKERRQ(ierr);
438  count++;
439  if (count % n_sum == 0) // output the time-averaged data
440  {
441  ierr = VecScale(dvec, 1.0 / count); CHKERRQ(ierr);
442  ierr = writeVec(dvec, t); CHKERRQ(ierr);
443  // reset for the next accumulation cycle
444  ierr = VecSet(dvec, 0.0); CHKERRQ(ierr);
445  count = 0;
446  }
447  }
448  else // output the time-step sub-volume data to file
449  {
450  ierr = writeVec(svec, t); CHKERRQ(ierr);
451  }
452  ierr = VecRestoreSubVector(fvec, is, &svec); CHKERRQ(ierr);
453 
454  PetscFunctionReturn(0);
